How I install Arch Linux

Lang: 
ping archlinux.org

timedatectl

cfdisk /dev/sdx

mkfs.ext4 /dev/root_partition
mkswap /dev/swap_partition
mkfs.fat -F 32 /dev/efi_system_partition

mkdir -p /mnt/boot/efi

mount /dev/rootpartition /mnt
mount /dev/efi_system_partition /mnt/boot/efi
swapon /dev/swap_partition

pacstrap -K /mnt base base-devel linux linux-firmware linux-headers sof-firmware nvim grub efibootmgr 
networkmanager plasma-desktop zsh reflector

genfstab -U /mnt >> /mnt/etc/fstab

arch-chroot /mnt

ln -sf /usr/shate/zoneinfo/Asia/Manila /etc/localtime

hwclock --systohc

edit /etc/locale.gen and uncomment en_US.UTF-8 UTF-8

Lang: 
locale-gen

Create locale.conf file

Lang: 
LANG=en_US.UTF-8

Create hostname file at /etc/hostname

Lang: 
yourhostname
Lang: 
passwd

useradd -m -G wheel -s /bin/zsh <name>
passwd <name>

EDITOR=nvim visudo

uncomment wheel

enable services

Lang: 
systemctl enable NetworkManager
systemctl enable sddm !or whatever login manager

! if on ssd
systemctl enable fstrim.timer
Lang: 
grub-install /dev/sdx
grub-mkconfig -o /boot/grub/grub.cfg

umount -a

install os-prober

edit /etc/default/grub and uncomment GRUB_DISABLE_OS_PROBER=false

grub-mkconfig -o /boot/grub/grub.cfg
